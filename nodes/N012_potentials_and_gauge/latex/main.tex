\section{Definitions}
\begin{align}
\mathbf{B} &= \nabla \times \mathbf{A},\\
\mathbf{E} &= -\nabla\phi - \frac{\partial \mathbf{A}}{\partial t}.
\end{align}

\section{Gauge Invariance}
\begin{align}
\mathbf{A}' &= \mathbf{A} + \nabla \Lambda,\\
\phi' &= \phi - \frac{\partial \Lambda}{\partial t}
\end{align}
leaves $\mathbf{E},\mathbf{B}$ invariant.

\section{Lorenz Gauge (Default)}
\begin{equation}
\frac{1}{c^2}\frac{\partial \phi}{\partial t} + \nabla\cdot\mathbf{A} = 0.
\end{equation}
In vacuum this yields homogeneous wave equations for $\phi$ and $\mathbf{A}$.
